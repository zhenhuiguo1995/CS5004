16    }
17  
18    public boolean checkItem(int purchaseNumber) {
19      return this.quantity >= purchaseNumber;
20    }
21  
22    /**
23     * Helper method to check if the product in stock is qualified to substitute the item in shopping
24     * cart.
25     *
26     * @param shoppingItem the product need to be substituted
27     * @return true if qualified, false if not
28     */
29    public boolean isQualifiedToSubstitute(Item shoppingItem) {
30      if (shoppingItem.getQuantity() > getQuantity()
31          || shoppingItem.getProduct().getPrice() < getProduct().getPrice()) {
32        return false;
33      }
34      if (shoppingItem.getProduct() instanceof AbstractGrocery) {
35        if (((AbstractGrocery) shoppingItem.getProduct()).getWeight()
36            > ((AbstractGrocery) (getProduct())).getWeight()) {
37          return false;
38        }
39      } else if (shoppingItem
40          .getProduct() instanceof AbstractHousehold) { // no need for the "if" here
41        if (((AbstractHousehold) shoppingItem.getProduct()).getUnits()
42            > ((AbstractHousehold) (getProduct())).getUnits()) {
43          return false;
44        }
45      }
46      return true;
47    }
48  }
