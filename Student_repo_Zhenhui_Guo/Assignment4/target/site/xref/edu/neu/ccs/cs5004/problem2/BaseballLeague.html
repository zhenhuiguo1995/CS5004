3   /**
4    * Represents detailed information about a baseball league.
5    */
6   public class BaseballLeague extends AbstractLeague{
7     private static final int ONE = 1;
8     private static final int BASEBALL_WIN_SCORE = 1;
9     private static final int BASEBALL_LOSE_SCORE = 0;
10  
11    /**
12     * Instantiates a new baseball league.
13     *
14     * @param leagueName the league name
15     * @param startMonths the start months
16     * @param endMonths the end months
17     * @param numberOfGames the number of games
18     * @param nextGame the next game
19     * @throws IllegalArgumentException the illegal argument exception
20     */
21    public BaseballLeague(String leagueName, int startMonths, int endMonths, int numberOfGames,
22        Game nextGame) throws IllegalArgumentException {
23      super(leagueName, startMonths, endMonths, numberOfGames, nextGame);
24    }
25  
26  
27    @Override
28    public AbstractLeague createGameForLeague(String leagueName, int startMonths, int endMonths,
29        int numberOfGames, Game game) {
30      return new BaseballLeague(leagueName, startMonths, endMonths, numberOfGames, game);
31    }
32  
33    @Override
34    public Game playGame(Game game, int homeTeamScore, int awayTeamScore) {
35      BaseballTeam homeTeam;
36      BaseballTeam awayTeam;
37      if (homeTeamScore > awayTeamScore) {
38        homeTeam = updateTeam(game, (BaseballTeam) game.getHomeTeam(), Result.WIN);
39        awayTeam = updateTeam(game, (BaseballTeam) game.getAwayTeam(), Result.LOSE);
40      } else {
41        homeTeam = updateTeam(game, (BaseballTeam) game.getHomeTeam(), Result.LOSE);
42        awayTeam = updateTeam(game, (BaseballTeam) game.getAwayTeam(), Result.WIN);
43      }
44      return new Game(homeTeam, awayTeam, game.getDate(), homeTeamScore, awayTeamScore);
45    }
46  
47    /**
48     * Update a baseball team based on the result of a game.
49     *
50     * @param game the game
51     * @param baseballTeam the baseball team
52     * @param result the result
53     * @return the baseball team
54     */
55    public BaseballTeam updateTeam(Game game, BaseballTeam baseballTeam, Result result) {
56      int gamePlayed = baseballTeam.getGamePlayed() + ONE;
57      int gameRemaining = baseballTeam.getGameRemaining() - ONE;
58      int points;
59      Record oldRecord = baseballTeam.getRecord();
60      Record newRecord;
61      if (result == Result.WIN) {
62        points = baseballTeam.getPoints() + BASEBALL_WIN_SCORE;
63        newRecord = new Record(oldRecord.getWins() + ONE, oldRecord.getLosses(),
64            oldRecord.getDraws());
65      } else {
66        points = baseballTeam.getPoints() + BASEBALL_LOSE_SCORE;
67        newRecord = new Record(oldRecord.getWins(), oldRecord.getLosses() + ONE,
68            oldRecord.getDraws());
69      }
70      return new BaseballTeam(baseballTeam.getName(), baseballTeam.getAbstractLeague(), gamePlayed,
71          gameRemaining, points, game, newRecord);
72    }
